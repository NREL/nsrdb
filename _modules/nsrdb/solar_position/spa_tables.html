    DELTA_T = DELTA_T.sort_index()

    def __getitem__(self, date):
        """Get the delta-t value for a given input date

        Parameters
        ----------
        date : datetime.date
            Date to retrieve delta t for.

        Returns
        -------
        delta_t : float
            Delta t value for input date
        """
        delta_t = None
        for i, row in self.DELTA_T.iterrows():
            if not pd.Timestamp(date) > i:
                delta_t = row['delta_t_tt-utc1']
                break
        return delta_t

    def __init__(self):
        resize_mapping = {
            'L1': (64, 3),
            'L2': (64, 3),
            'L3': (64, 3),
            'L4': (64, 3),
            'L5': (64, 3),
            'B1': (5, 3),
            'R1': (40, 3),
            'R2': (40, 3),
            'R3': (40, 3),
            'R4': (40, 3),
        }

        # make arrays uniform size for efficient broadcasting, fill with 0s
        for key, dims in resize_mapping.items():
            new_rows = dims[0] - self.TABLE_LBR_DICT[key].shape[0]
            self.TABLE_LBR_DICT[key] = np.append(
                self.TABLE_LBR_DICT[key], np.zeros((new_rows, 3)), axis=0
            )

    @property
    def helio_long_table(self):
        """
        Generation heliocentric longitude table

        Returns
        -------
        helio_long_table : ndarray
            Heliocentric longitude table
        """

            [
                self.TABLE_LBR_DICT['L0'],

                self.TABLE_LBR_DICT['L2'],
                self.TABLE_LBR_DICT['L3'],
                self.TABLE_LBR_DICT['L4'],
